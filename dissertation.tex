% Here is a sample format for dissertation in economics.  You should check
% the University of Pennsylvania Doctoral Dissertation Manual to adjust
% any changes they made at the webpage:

% http://guides.library.upenn.edu/dissertation_manual/formatting

\documentclass[11pt]{report}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%%%%%% Packages %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\usepackage{amssymb}
\usepackage{amsmath}
\usepackage{amsthm}
\usepackage[nottoc]{tocbibind}
\usepackage[toc,page]{appendix}
\usepackage{booktabs}
\usepackage{titlesec}
\usepackage{chngcntr}
\usepackage{graphicx}

% using biblatex and biber instead of natbib
\usepackage[backend=biber,
            style=authoryear,
            doi=false,url=false,isbn=false,
            maxcitenames=2]{biblatex}
\addbibresource{references.bib}
\AtEveryBibitem{\clearlist{language}} % clears language from bib entries
\usepackage{csquotes}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%%%%%% Document setup %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

% margins
\usepackage[letterpaper,
            nohead,
            includefoot,
            top=1in,
            bottom=1in,
            left=1.5in,
            right=1in]{geometry}

\usepackage[doublespacing]{setspace} % set linespacing

% numbering format
\counterwithout{footnote}{chapter}
\titleformat{\section}
{\normalfont\fontsize{12}{15}\bfseries}{\thesection}{1em}{}

\titleformat{\subsection}
{\normalfont\fontsize{12}{15}\bfseries}{\thesubsection}{1em}{}

\numberwithin{equation}{section}

% page style
\pagestyle{plain}


%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%%%%%% Definitions %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\def\thetitle{TITLE}
\def\theauthor{Name}
\def\theadvisor{Advisor}
\def\THEAUTHOR{NAME}
\def\THEADVISOR{ADVISOR}
\def\theyear{2018}


\begin{document}

% title
\input{frontmatter/title}
\newpage

% copyright
\input{frontmatter/copyright}
\newpage

\doublespacing % for rest of frontmatter

% dedication
\input{frontmatter/dedication}
\newpage

% acknowledgment
\input{frontmatter/acknowledgement}
\newpage

% abstract
\input{frontmatter/abstract}
\newpage

\singlespacing % for toc and lists

% table of contents
\tableofcontents
\newpage

% list of tables
\listoftables
\newpage

% list of figures
\listoffigures
\newpage

\doublespacing % for main text
\pagenumbering{arabic}

\chapter*{Introduction}
\addcontentsline{toc}{chapter}{Introduction}
\input{content/introduction}
\newpage

\chapter{Title of the chapter}
\input{content/chapter1}
\newpage

\chapter{Title of the chapter}
\input{content/chapter2}
\newpage

% appendices
\input{content/appendix}
\newpage

% bibliography
\printbibliography[heading=bibintoc, title={Bibliography}]

\end{document}
